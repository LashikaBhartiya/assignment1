# Program to take two numbers as input and print their sum

# Taking first number as input from the user
number1 = float(input("Enter the first number: "))

# Taking second number as input from the user
number2 = float(input("Enter the second number: "))

# Calculating the sum of the two numbers
sum_result = number1 + number2

# Printing the sum
print("The sum of", number1, "and", number2, "is:", sum_result)
